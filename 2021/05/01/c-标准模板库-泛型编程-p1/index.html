STL(标准模板库)


STL是一套功能强大的 C++ 模板类，提供了通用的模板类和函数，这些模板类和函数可以实现多种流行和常用的算法和数据结构，如向量、链表、队列、栈

基本分为三个部分容器、算法、迭代器


容器


容器是用来管理某一类对象的集合，C++ 提供了各种不同类型的容器，比如 deque、list、vector、map 等

在之前已经大概了解了vector的基本用法和原理，vector也可以用来进行动态内存分配

vector在前面说过，是自动使用new和delete（可以使用动态内存），如果忽略模板参数的值，会由自动使用分配器


#include <iostream>
#include <vector>
using namespace std;

const int N = 10;

int main(){
    vector<int> pdata(5);
    vector<int> pd(N);//支持动态内存分配

    pdata[0] = 1;//类中的[]运算符是有重载过，可以使用该索引方法
    cout << pdata[0] << endl;

    return 0;
}

迭代器


迭代器用于遍历对象集合的元素。这些集合可能是容器，也可能是容器的子集。你可以把迭代器当作指针（在实际使用中的语法跟指针类似），一些特殊的方法对集合的进行判断

迭代器为不同容器类提供统一接口，方便程序的操作


#include <iostream>
#include <vector>
using namespace std;

int main(){
    vector<int> pd(5);
    int i;
    for (i = 0; i < 5; i ++){
        pd.push_back(i);//push_back( ) 成员函数在向量的末尾插入值，如果有必要会扩展向量的大小
    }
    
    //vector<int>::iterator idx = pd.begin();
    auto idx = pd.begin();//c++11 新增了自动类型推断，可以不用像上面一样
    while (idx != pd.end()){
        cout << *idx << endl;
        idx++;
    }

    return 0;
}

STL的其他操作


在程序中我们一般要对多个数组进行操作，包括搜索、排序等等。STL正是简化了这种重复性高的工作，假设有8个容器，需要支持10种操作，一般需要定义80个成员函数。如果采用STL方法，只需要定义10个非成员函数来进行这种操作

STL中拥有各种简便的方法

for_each：可以用这种代替反复枯燥的for循环


#include <iostream>
#include <vector>
#include <algorithm>//STL存在的头文件
using namespace std;

void print(const int & x);

int main(){
    int a[] = {1, 2, 3, 4, 5};
    //siezeof(a) = N * sizeof(int) N为元素个数
    vector<int> v(a, a + sizeof(a) / sizeof(int));//通过数组地址对vector进行初始化
    //vector<int> v(a, a + 5); 这样初始化也可以，前提是数组元素比较小且不变动

    for_each(v.begin(), v.end(), print);//开始，结束，使用的函数
    //可以跳过使用迭代器之后进行的一大堆for循环的编程

    return 0;
}

void print(const int & x){
    cout << x << endl;
}

sort：之前就已经介绍过了，黄老板已经给我介绍的一句话排序

random_shuffle：将容器中的元素随机打乱，需要能过随机访问值的容器，vector就十分合适


#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void print(const int & x);
bool judge(const int & x1, const int & x2);

int main(){
    int a[] = {1, 2, 3, 4, 5, 6};
    vector<int> v(a, a + sizeof(a) / sizeof(int));
    
    random_shuffle(v.begin(), v.end());
    for_each(v.begin(), v.end(), print);

    cout << endl;
    sort(v.begin(), v.end(), judge);//自定义判断标准
    for_each(v.begin(), v.end(), print);

    return 0;
}

void print(const int & x){
    cout << x << endl;
}

bool judge(const int & x1, const int & x2){
    return x1 > x2;
}

基于范围的for循环


c++11中新增的for循环（可以理解为Java中的foreach循环），为STL设计，将for_each进一步简化

在这个c++20即将推出之际，可以看到，各种语言也是在为了便利性不断地向前延申


#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void print(const int & x);

int main(){
    int a[] = {1, 2, 3, 4, 5, 6};
    vector<int> v(a, a + sizeof(a) / sizeof(int));

    for(auto & i : v){//新增的auto判断可以省略判断类型的过程"int i"
        print(i);
    }

    return 0;
}

void print(const int & x){
    cout << x << endl;
}

泛型编程介绍


在使用泛型编程之前，我们先回顾一下c链表的操作（基本的功能无非就是增删改查）

（也算是看看现在我能不能看懂链表（x）


#include <iostream>
using namespace std;

struct Node{
    double item;
    Node * p_next;
};
typedef struct Node Node;

Node *create();
Node *num_sort(Node * head);
void destroy(Node * head);
void print(Node *head);

int main(){
    Node *head;
    head = create();

    cout << "origin version" << endl;
    print(head);

    head = num_sort(head);
    cout << "sorted version" << endl;
    print(head);

    destroy(head);

    return 0;
}

Node *create(){
    Node *p, *head, *tail;
    int num;
    head = NULL;
    tail = NULL;
    
    cout << "enter a num (end with -999)" << endl;
    cin >> num;
    while(num != -999){
        p = new Node;
        p->item = num;
        p->p_next = NULL;

        if(head == NULL){
            head = p;
        }else{
            tail->p_next = p;
        }
        tail = p;

        cout << "enter a num (end with -999)" << endl;
        cin >> num;
    }

    return head;
}

Node *num_sort(Node * head){
    Node *p1, *p2, temp;
    p1 = head;

    if(head == NULL){
        cout << "It's NULL" << endl;
        return head;
    }else if(head->p_next == NULL){
        cout << "only one number, no need to sort" << endl;
        return head;
    }

    while(p1->p_next != NULL){
        p2 = p1->p_next;
        while(p2 != NULL){
            if(p1->item > p2->item){
                temp = *p1;
                *p1 = *p2;
                *p2 = temp;
                temp.p_next = p1->p_next;
                p1->p_next = p2->p_next;
                p2->p_next = temp.p_next;
            }
            p2 = p2->p_next;
        }
        p1 = p1->p_next;
    }

    return head;
}

void print(Node *head){
    Node *p;
    p = head;

    if(head == NULL){
        cout << "nothing to print" << endl;
    }else{
        cout << "num is " << endl;
        while(p != NULL){
            cout << p->item << endl;
            p = p->p_next;
        }
    }
}

void destroy(Node * head){
    Node *p1, *p2;
    p1 = head;

    while(p1 != NULL){
        p2 = p1;
        p1 = p1->p_next;
        delete p2;
    }

    cout << "release success" << endl;
}

泛型编程旨在使用同一个find函数处理数组和链表问题，于是就产生了各种类型的迭代器

输入迭代器：这种迭代器必须能够访问容器中的所有值（原理是根据++运算符来实现的）。并且，不能保证两次遍历容器时都是相同的顺序，也不能保证先前的值仍然可以被解除引用

（输入迭代器是单向迭代器，可以递增不能倒退）

输出迭代器：输出迭代器和输入迭代器类似，解除引用让程序能够修改容器值

正向迭代器：正向迭代器只使用++运算符，每次遍历数组时都是使用同一个顺序。递增之后，仍然可以对前面的迭代器解除引用，并可以得到相同的值

双向迭代器：显而易见，支持++运算符和–运算符

随机访问迭代器：有些算法要求能够直接跳到容器中的任何一个元素


迭代器实践


之前在例子中已经介绍过了begin()&end()这两种迭代器，vector是动态内存，所以需要这些机制来扩充自己

front()：返回当前vector容器中起始元素的引用

back()：返回当前vector容器中末尾元素的引用

copy(start, end, container.begin()): 把一个序列拷贝到一个容器中去


#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int SIZE = 3;

int main(){

    int a[SIZE] = {3, 4, 5};
    int b[SIZE];
    copy(a, a + SIZE, b);

    //可以看见，基于范围的for循环还是将值赋值给i
    for(auto i : b){
        cout << i << endl;//不需要使用函数，直接通过这种方式也可以输出数组的值
    }

    vector<int> temp(SIZE);
    copy(a, a + SIZE, &temp.front());//要注意，front()的返回值是引用
    for(auto i : temp){
        cout << i << endl;
    }

    return 0;
}

输入迭代器


istream_iterator 定义在<iterator>头文件，并位于 std 命名空间中

创建输入流迭代器的方式有 3 种

调用 istream_iterator 模板类的默认构造函数

创建一个可用来读取数据的输入流迭代器，std::istream_iterator<double> iit(std::cin);

支持用已创建好的 istream_iterator 迭代器为新建 istream_iterator 迭代器初始化，std::istream_iterator<double> iit2(iit1);


#include <iostream>
#include <iterator>
#include <algorithm>
using namespace std;

int main(){
    double v1, v2;
    int i;
    istream_iterator<double> eos;//创建表示结束的输入迭代器
    istream_iterator<double> lit(cin);//创建一个可逐个读取输入流中数据的迭代器，同时这里会让用户输入数据

    if(lit != eos){
        v1 = *lit;
    }
    lit++;
    if(lit != eos){
        v2 = *lit;
    }

    cout << "value1= " << v1 << endl;
    cout << "value2= " << v2 << endl;

    return 0;
}

注意，只有读取到 EOF 流结束符时，程序中的 iit 才会和 eos 相等

在这个例子中，输入迭代器会自动根据要赋值的个数决定用户输入的次数

但我们大部分情况下会使用while循环进行赋值，这样就没办法通过自动判断的方法来中止了。Windows 平台上使用 Ctrl+Z 组合键输入 ^Z 表示 EOF 流结束符，此结束符需要单独输入，或者输入换行符之后再输入才有效


#include <iostream>
#include <vector>
#include <iterator>
#include <algorithm>
using namespace std;

int main(){
    vector<double> v;
    int i;
    istream_iterator<double> eos;
    istream_iterator<double> lit(cin);
    
    while(lit != eos){
        v.push_back(*lit);
        lit++;
    }//停止输入需要手动输入^Z

    for(auto j : v){
        cout << j << " ";
    }
    
    return 0;
}

输出迭代器


ostream_iterator 模板类中也提供了 3 种创建 ostream_iterator 迭代器的方法

通过调用该模板类的默认构造函数

在第一种方式的基础上，还可以为写入的元素之间指定一个分隔符，std::ostream_iterator<int> out_it(std::cout，",");

在创建输出流迭代器时，可以用已有的同类型的迭代器，为其初始化，std::ostream_iterator<int> out_it1(out_it);


#include <iostream>
#include <iterator>
#include <algorithm>
#include <string>
using namespace std;

int main(){
    ostream_iterator<string> out(cout);
    *out = "http://c.biancheng.net/view/7286.html";
    cout << endl;

    ostream_iterator<int> out_iterator(cout, ",");
    *out_iterator = 1;
    *out_iterator = 2;
    *out_iterator = 3; 
    //轮流输入之后，输出迭代器会自动将指针递增

    return 0;    
}

和输入迭代器一样，在一般情况下，我们是不会这么简单地直接赋值地，大部分情况都需要使用第三者

由于不能通过参数，所以就需要借用copy()


#include <iostream>
#include <algorithm>
#include <iterator>
#include <vector>
using namespace std;

int main(){
    vector<int> v;
    ostream_iterator<int> ov(cout, " ");
    //ostream_iterator<int, char> ov(cout, " "); 写成这种类型也可以
    int num;

    cin >> num;
    while(num != -999){
        v.push_back(num);
        cin >> num;
    }

    sort(v.begin(), v.end());
    copy(v.begin(), v.end(), ov);

    return 0;
}
