在初始化下拉表框时，可以选择同时指定下拉列表中的项目内容


下拉列表框的项目封装为ComboBoxModel的情况比较多，可以在初始化对象时调用这个接口

调用接口的同时，必须实现以下两种方法

public void setSelectedItem(Object item)

public Object getSelectedItem() 用于返回下拉列表种的选中项

getSize()

getElementAt(int index) 返回指定索引处的值


import javax.swing.*;
import java.awt.*;

public class JComboBoxModelTest extends JFrame {

    JComboBox<String> jComboBox = new JComboBox<>(new MyComboBox());//应用了JDK7的新特性
    JLabel jLabel = new JLabel("请选择证件");

    public JComboBoxModelTest(){

        Container container = getContentPane();
        setTitle("下拉列表组件");
        container.setLayout(new FlowLayout());
        container.add(jLabel);
        container.add(jComboBox);
        setVisible(true);
        setSize(1000, 1000);
        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);

    }

    public static void main(String[] args) {

        new JComboBoxModelTest();

    }

}

class MyComboBox extends AbstractListModel<String> implements ComboBoxModel<String>{

    String selecteditem = null;
    String[] test = {"身份证", "军人证", "学生证", "工作证"};

    @Override
    public String getElementAt(int index) {//根据索引值返回值

        return test[index];

    }

    @Override
    public int getSize() {

        return test.length;

    }

    @Override
    public void setSelectedItem(Object item) {//设置下拉列表框项目

        selecteditem = (String) item;

    }

    @Override
    public Object getSelectedItem() {//获取下拉列表框种的项目

        return selecteditem;

    }

}

文本框组件


文本框用来显示或编辑一个单行文本，所以设置参数的时候只能设定对应的宽度

public JTextField()

public JTextField(Document docModel, String text, int fieldWidth)


import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class JTextFieldTest extends JFrame {

    public JTextFieldTest(){

        final JTextField jTextField = new JTextField("aaa", 20);
        final JButton jButton = new JButton("清除");

        setLayout(new FlowLayout());//使用流式排序
        Container container = getContentPane();
        container.add(jTextField);
        container.add(jButton);
        jTextField.setVisible(true);
        setTitle("文本域的练习");
        setSize(100, 100);
        setVisible(true);
        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);

        jButton.addActionListener(new ActionListener() {

            @Override
            public void actionPerformed(ActionEvent e) {

                jTextField.setText("");//通过设置为“无”，将文本框置空
                jTextField.requestFocus();//焦点回归到文本框

            }

        });

    }

    public static void main(String[] args) {

        new JTextFieldTest();
    }

}

事件监听器


在Swing程序中，通常有三个对象，分别完成事情的处理

1.事件源 2.事件 3.监听器触发对应事件

动作事件就会使用ActionEvent类对象，这个类对应的接口是ActionListener

public addActionListener()

public removeActionListener()


动作监视器


import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class ListenerTest extends JFrame{

    private JButton jButton = new JButton("点我有惊喜");

    public ListenerTest(){

        Container container = getContentPane();
        container.add(jButton);
        container.setLayout(new FlowLayout());

        setTitle("动作监视器的练习");
        setVisible(true);
        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
        setSize(1000, 1000);

        jButton.addActionListener(new ActionListener() {

            @Override
            public void actionPerformed(ActionEvent e) {

                jButton.setText("ヽ(✿ﾟ▽ﾟ)ノ");

            }

        });

    }
    

    public static void main(String[] args) {

        new ListenerTest();

    }

}
